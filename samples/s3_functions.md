// Sample Program 3: Function Definition and Application
let add = fn(a, b) => a + b
let multiply = fn(x, y) => x * y
let greet = fn(person) => "Hello, " + person

let result1 = add(5, 3)
let result2 = multiply(4, 7)
let message = greet("Alice")

// This program showcases:
// - Function keyword: fn
// - Function operator: =>
// - Delimiters: (, ), ,
// - Function parameters: a, b, x, y, person
// - Function application with parentheses and commas
// - Mixed data types in function operations
// - Identifiers: add, multiply, greet, result1, result2, message
